15package org.tp23.antinstaller.runtime.logic;
16
17import java.util.Comparator;
18
19import org.apache.tools.ant.taskdefs.Length;
20
21/**
22 * Compare strings based on their length. Used by ExpressionBuilder to ensure
23 * that longest match is attempted first.
24 * 
25 * @author mwilson
26 * @version $Id: StringLengthComparator.java,v 1.1 2006/09/08 19:16:31
27 *          anothermwilson Exp $
28 * @since 0.7.4 patch 2
29 */
30
31public class StringLengthComparator implements Comparator {
32
33    /**
34     * Sort by descending String length
35     * 
36     * @param o1 first string
37     * @param o2 second string
38     * @return difference between the length of o2 and o1
39     */
40    public int compare(Object o1, Object o2) {
41        String str1 = (String) o1;
42        String str2 = (String) o2;
43        return (str2.length() - str1.length());
44    }
45    
46}
