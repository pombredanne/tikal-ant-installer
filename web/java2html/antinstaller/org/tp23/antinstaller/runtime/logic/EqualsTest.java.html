16
17/**
18 * @author mwilson
19 * @version $Id: EqualsTest.java,v 1.2 2006/12/21 00:03:18 teknopaul Exp $
20 * @since 0.7.4 patch 2
21 */
22public class EqualsTest extends ValuesTest {
23
24    private static final String[] TEST_TOKENS = { "==", "=" };
25
26    public String[] getTestTokens() {
27        return TEST_TOKENS;
28    }
29
30    protected boolean getTestResult(final String propValue, final String expectedValue) {
31        // Where the expected value is null, empty string must also match
32        if (expectedValue == null) {
33            return (propValue == null) || propValue.equals("");
34        }
35
36        if ((propValue != null) && propValue.equals(expectedValue)) {
37            return true;
38        }
39
40        return false;
41    }
42}
