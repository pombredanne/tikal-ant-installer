16package org.tp23.antinstaller.runtime;
17
18import org.tp23.antinstaller.InstallerContext;
19import org.tp23.antinstaller.antmod.RuntimeLauncher;
20import org.tp23.antinstaller.page.Page;
21
22/**
23 * Abstract Runner superclass that handles the runPost(page) to execute
24 * ant tasks mid display.
25 * @author teknopaul
26 */
27public abstract class AntRunner implements Runner{
28
29    private RuntimeLauncher launcher = null;
30    private InstallerContext ctx;
31    
32    public AntRunner(InstallerContext ctx) {
33        this.ctx = ctx;
34    }
35
36    protected void runPost(Page page){
37        String postTarget = page.getPostDisplayTarget();
38        if(postTarget != null){
39            if(launcher == null){
40                prepareLauncher();
41            }
42            launcher.updateProps();
43            launcher.run(postTarget);
44        }
45    }
46    /**
47     * This should never get run if there are no postTargets
48     *
49     */
50    private void prepareLauncher(){
51        launcher = new RuntimeLauncher(ctx);
52        launcher.parseProject();
53    }
54}
