18import org.tp23.antinstaller.InstallException;

20
21/**
22 * <p>A Runtime exception that implies the config file is wrong.</p>
23 */
24public class ConfigurationException
25    extends InstallException {
26    public ConfigurationException() {
27        super();
28    }
29
30    public ConfigurationException(String message) {
31        super(message);
32    }
33
34    public ConfigurationException(String message, Throwable cause) {
35        super(message, cause);
36    }
37
38}
