16package org.tp23.antinstaller.renderer.text;
17
18import java.io.BufferedReader;
19import java.io.IOException;
20import java.io.PrintStream;
21import java.util.ResourceBundle;
22
23import org.tp23.antinstaller.InstallerContext;
24import org.tp23.antinstaller.input.CheckboxInput;
25import org.tp23.antinstaller.input.InputField;
26import org.tp23.antinstaller.input.OutputField;
27
28
29public class CheckboxInputRenderer
30    implements TextOutputFieldRenderer {
31
32    private static final ResourceBundle res = ResourceBundle.getBundle("org.tp23.antinstaller.renderer.text.Res");
33
34    protected InstallerContext ctx;
35
36    public CheckboxInputRenderer() {
37    }
38
39    public void setContext(InstallerContext ctx) {
40        this.ctx = ctx;
41    }
42
43    public void renderOutput(OutputField field, BufferedReader reader, PrintStream out) throws IOException {
44        CheckboxInput iField = (CheckboxInput) field;
45        out.print(iField.getDisplayText());
46        out.print("   [");
47        out.print(res.getString("_default_"));
48        out.print(":");
49        out.print(iField.getDefaultValue());
50        out.print("]");
51        if (InputField.isTrue(iField.getForce())) {
52            out.println(res.getString("_required_"));
53            iField.setValue(iField.getDefaultValue());
54            return;
55        }
56        else {
57            out.println();
58        }
59
60
61        String input = reader.readLine();
62        out.println();
63        if (input == null || input.trim().equals("")) {
64            input = iField.getDefaultValue();
65        }
66        if(InputField.isTrue(input)){
67            iField.setValue("true");
68        }
69        //@TODO should accept true or false only and complain toher wise looping till sensible input
70        //else if (InputField.isFalse(input)){
71        //  iField.setValue("false");
72        //}
73        else{
74            iField.setValue("false");
75        }
76    }
77
78    public boolean isAbort() {
79        return false;
80    }
81
82
83
84    /**
85     * renderError
86     *
87     * @param field InputField
88     * @param in InputStream
89     * @param out PrintStream
90     */
91    public void renderError(OutputField field, BufferedReader reader, PrintStream out) {
92    }
93}
