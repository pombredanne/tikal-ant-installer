16package org.tp23.antinstaller.renderer;
17
18import org.tp23.antinstaller.InstallerContext;
19/**
20 *
21 * <p>Abstract message renderer asking user questions or providing feedback</p>
22 * <p> </p>
23 * <p>Copyright: Copyright (c) 2004</p>
24 * <p>Company: tp23</p>
25 * @author Paul Hinds
26 * @version $Id: MessageRenderer.java,v 1.1.1.1 2005/10/18 18:20:59 teknopaul Exp $
27 */
28public interface MessageRenderer {
29
30    public void setInstallerContext(InstallerContext ctx);
31    /**
32     * Print a message for which there is not option to reply
33     * @param message String
34     */
35    public void printMessage(String message);
36    /**
37     * ask the user for Yes or No
38     * @param message String
39     * @return boolean  true = Yes  false = No
40     */
41    public boolean prompt(String message);
42}
