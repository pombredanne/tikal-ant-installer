16package org.tp23.antinstaller.renderer.swing;
17
18import java.awt.Color;
19
20import org.tp23.antinstaller.input.ExtValidatedTextInput;
21import org.tp23.antinstaller.input.Validator;
22import org.tp23.antinstaller.renderer.MessageRenderer;
23
24public class ExtValidatedTextInputRenderer
25    extends ValidatedTextInputRenderer {
26
27    /**
28     * renderError
29     */
30    public void renderError() {
31        MessageRenderer mr = ctx.getMessageRenderer();
32        ExtValidatedTextInput extVal = (ExtValidatedTextInput)inputField;
33        Validator validator = extVal.getValidator();
34        Throwable t = extVal.getThrowable();
35        mr.printMessage(validator.getErrorMessage(t, null));
36        this.jTextField.requestFocus();
37        this.jTextField.setForeground(Color.red);
38    }
39
40}
