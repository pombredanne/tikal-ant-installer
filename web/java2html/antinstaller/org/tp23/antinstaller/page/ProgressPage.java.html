16package org.tp23.antinstaller.page;
17
18import org.tp23.antinstaller.input.OutputField;
19
20public class ProgressPage extends Page{
21    
22    private boolean showTargets = true;
23    
24    
25    public ProgressPage() {
26    }
27    /**
28     * @return Returns the showTargets.
29     */
30    public boolean isShowTargets() {
31        return showTargets;
32    }
33    /**
34     * @param showTargets indicates that the graphical display of
35     * progress should be used in the Swing renderer
36     */
37    public void setShowTargets(boolean showTargets) {
38        this.showTargets = showTargets;
39    }
40    public void setShowTargets(String strShowTargets) {
41        this.showTargets = OutputField.isTrue(strShowTargets);
42    }
43}
